   201→// ── Financial helper ─────────────────────────────────────────────────────────
   202→// Derives a deterministic price from the order so no schema change is needed.
   203→// Model: customer pays orderValue → 75% print centre · 15% OEM royalty · 10% platform fee
   204→export function getOrderFinancials(order: Order) {
   205→  const seed = order.id.split('').reduce((acc, c) => acc + c.charCodeAt(0), 0)
   206→  const unitPrice = Math.round((1800 + (seed % 6200)) / 100) * 100   // $1 800–$8 000
   207→  const orderValue = unitPrice * order.quantity
   208→  return {
   209→    orderValue,
   210→    printCost:   Math.round(orderValue * 0.75),   // → Print Centre
   211→    oemRoyalty:  Math.round(orderValue * 0.15),   // → OEM (IP licence fee)
   212→    platformFee: Math.round(orderValue * 0.10),   // → AddManuChain
   213→  }
   214→}
