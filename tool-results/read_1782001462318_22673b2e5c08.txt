   197→function genId(prefix: string) {
   198→  return `${prefix}-${Date.now().toString(36)}${Math.random()
   199→    .toString(36)
   200→    .slice(2, 6)}`
   201→}

   203→function timeAgo(iso: string): string {
   204→  try {
   205→    const diff = Date.now() - new Date(iso).getTime()
   206→    const s = Math.floor(diff / 1000)
   207→    if (s < 60) return `${s}s ago`
   208→    const m = Math.floor(s / 60)
   209→    if (m < 60) return `${m}m ago`
   210→    const h = Math.floor(m / 60)
   211→    if (h < 24) return `${h}h ago`
   212→    const d = Math.floor(h / 24)
   213→    return `${d}d ago`
   214→  } catch {
   215→    return '—'
   216→  }
   217→}
   218→
   219→function fmtTime(iso: string | null): string {
   220→  if (!iso) return '—'
   221→  try {
   222→    return new Date(iso).toLocaleTimeString([], {
   223→      hour: '2-digit',
   224→      minute: '2-digit',
   225→    })
   226→  } catch {
   227→    return '—'
   228→  }
   229→}
   230→
   231→function safeParse<T>(raw: string | null | undefined, fallback: T): T {
   232→  if (!raw) return fallback
   233→  try {
   234→    return JSON.parse(raw) as T
   235→  } catch {
   236→    return fallback
   237→  }
   238→}

   287→  const pct = Math.round((confidence ?? 0) * 100)
   288→  const color =
   289→    pct >= 80
   290→      ? 'border-emerald-200 bg-emerald-50 text-emerald-700'

   294→  return (
   295→    <Badge variant="outline" className={color}>
   296→      <CircleDot className="h-3 w-3" />
   297→      {pct}% confidence
   298→    </Badge>

  1025→export function AIAgentConsole() {
  1026→  const [messages, setMessages] = useState<ChatMessage[]>([])
  1027→  const [input, setInput] = useState('')
  1028→  const [sending, setSending] = useState(false)
  1029→  const [approvingId, setApprovingId] = useState<string | null>(null)
  1030→  const [logs, setLogs] = useState<AuditLog[]>([])
  1031→  const [logsLoading, setLogsLoading] = useState(false)
  1032→
  1033→  const threadRef = useRef<HTMLDivElement>(null)
  1034→  const inputRef = useRef<HTMLTextAreaElement>(null)
  1035→
  1036→  // ── Fetch audit feed ─────────────────────────────────────────────────────
  1037→  const fetchLogs = useCallback(async () => {
  1038→    setLogsLoading(true)
  1039→    try {
  1040→      const res = await fetch('/api/ai/agent?limit=20', { cache: 'no-store' })
  1041→      if (!res.ok) throw new Error(`HTTP ${res.status}`)
  1042→      const json = await res.json()
  1043→      setLogs((json.logs as AuditLog[]) ?? [])
  1044→    } catch (e: any) {
  1045→      // Silent — the audit feed is secondary
  1046→      console.error('audit feed fetch failed', e)
  1047→    } finally {
  1048→      setLogsLoading(false)
  1049→    }
  1050→  }, [])
  1051→
  1052→  useEffect(() => {
  1053→    fetchLogs()
  1054→  }, [fetchLogs])
  1055→
  1056→  // ── Auto-scroll to bottom when new messages arrive ───────────────────────
  1057→  useEffect(() => {
  1058→    if (threadRef.current) {
  1059→      threadRef.current.scrollTop = threadRef.current.scrollHeight
  1060→    }
  1061→  }, [messages])
  1062→
  1063→  // ── Send a new request ───────────────────────────────────────────────────
  1064→  const sendRequest = useCallback(
  1065→    async (text: string) => {
  1066→      const trimmed = text.trim()
  1067→      if (!trimmed || sending) return
  1068→
  1069→      const userMsgId = genId('u')
  1070→      const agentMsgId = genId('a')
  1071→      const userMsg: ChatMessage = {
  1072→        id: userMsgId,
  1073→        role: 'user',
  1074→        text: trimmed,
  1075→      }
  1076→      const thinkingMsg: ChatMessage = {
  1077→        id: agentMsgId,
  1078→        role: 'agent',
  1079→        thinking: true,
  1080→      }
  1081→      setMessages((prev) => [...prev, userMsg, thinkingMsg])
  1082→      setInput('')
  1083→      setSending(true)
  1084→
  1085→      try {
  1086→        const res = await fetch('/api/ai/agent', {
  1087→          method: 'POST',
  1088→          headers: { 'Content-Type': 'application/json' },
  1089→          body: JSON.stringify({ request: trimmed, autoExecute: false }),
  1090→        })
  1091→        const json = (await res.json()) as AgentResponse
  1092→        if (!res.ok) {
  1093→          throw new Error(json?.error ?? `HTTP ${res.status}`)
  1094→        }
  1095→        setMessages((prev) =>
  1096→          prev.map((m) =>
  1097→            m.id === agentMsgId
  1098→              ? {
  1099→                  ...m,
  1100→                  thinking: false,
  1101→                  response: json,
  1102→                }
  1103→              : m,
  1104→          ),
  1105→        )
  1106→        // Refresh audit feed so the new action shows up
  1107→        fetchLogs()
  1108→      } catch (e: any) {
  1109→        const errResp: AgentResponse = {
  1110→          sessionId: 'sess-error',
  1111→          request: trimmed,
  1112→          plan: {
  1113→            intent: 'unknown',
  1114→            tool: 'unknown',
  1115→            parameters: {},
  1116→            confidence: 0,
  1117→            requiresApproval: false,
  1118→            reasoning: 'The agent request failed.',
  1119→            userFacingSummary: 'Request failed.',
  1120→          },
  1121→          autoExecuted: false,
  1122→          status: 'failed',
  1123→          error: e?.message ?? 'Unknown error',
  1124→        }
  1125→        setMessages((prev) =>
  1126→          prev.map((m) =>
  1127→            m.id === agentMsgId
  1128→              ? { ...m, thinking: false, response: errResp }
  1129→              : m,
  1130→          ),
  1131→        )
  1132→        toast.error('Agent request failed', {
  1133→          description: e?.message ?? 'Unknown error',
  1134→        })
  1135→      } finally {
  1136→        setSending(false)
  1137→      }
  1138→    },
  1139→    [sending, fetchLogs],
  1140→  )
  1141→
  1142→  // ── Approve an awaiting-approval action ──────────────────────────────────
