     6→const rateLimitMap = new Map<string, { count: number; resetAt: number }>()
     7→function checkRateLimit(ip: string): boolean {
     8→  const now = Date.now()
     9→  const entry = rateLimitMap.get(ip)
    10→  if (!entry || now > entry.resetAt) {
    11→    rateLimitMap.set(ip, { count: 1, resetAt: now + 60_000 })
    12→    return true
    13→  }
    14→  if (entry.count >= 20) return false
    15→  entry.count++
    16→  return true
    17→}

   386→async function executeTool(
   387→  name: string,
   388→  args: Record<string, unknown>,
   389→  baseUrl: string,
   390→): Promise<{ result: string; frontendAction?: Record<string, unknown> }> {

   395→      const page = args.page as string
   396→      const labels: Record<string, string> = {
   397→        overview: 'Dashboard Overview', orders: 'Orders', print_queue: 'Print Queue',
   398→        blueprints: 'Blueprint Library', centers: 'Print Centers', peer_printers: 'Peer Printers',
   399→        shipments: 'Shipments', materials: 'Materials', partners: 'OEM Partners',
   400→        analytics: 'Analytics', audit: 'Audit Logs', certifications: 'Certifications',
   401→        authorities: 'Certification Authorities', services: 'Customer Success',
   402→        digital_inventory: 'Digital Inventory', physical_inventory: 'Physical Inventory',
   403→        ip_library: 'IP Library', lab_portal: 'Lab & Testing Portal', settings: 'Settings',
   404→        emergency: 'Emergency Response', oem_portal: 'OEM Self-Service Portal',
   405→        cooperative: 'Digital Cooperative', material_properties: 'Material Properties Library',
   406→        feasibility: 'AM Feasibility Triage', sc_intelligence: 'Supply Chain Intelligence',
   407→      }
   408→      return {
   409→        result: `Navigated to ${labels[page] ?? page}`,
   410→        frontendAction: { type: 'navigate', page, label: labels[page] ?? page },
   411→      }
   412→    }
   413→
   414→    case 'get_dashboard_stats': {
   415→      try {
   416→        const res = await fetch(`${baseUrl}/api/stats`)
   417→        const data = await res.json()
   418→        return { result: JSON.stringify(data) }
   419→      } catch {
   420→        return { result: 'Could not fetch stats at this time.' }
   421→      }
   422→    }
   423→
   424→    case 'list_orders': {
   425→      try {
   426→        const params = new URLSearchParams()
   427→        if (args.status) params.set('status', args.status as string)
   428→        const res = await fetch(`${baseUrl}/api/orders?${params}`)
   429→        const data = await res.json()
   430→        const orders = (Array.isArray(data) ? data : data.orders ?? [])
   431→          .slice(0, (args.limit as number) ?? 5)
   432→        if (orders.length === 0) return { result: 'No orders found matching the criteria.' }
   433→        const summary = orders.map((o: Record<string, unknown>) =>
   434→          `#${o.id ?? o.orderId} — ${o.partName ?? o.name} | Status: ${o.status} | Priority: ${o.priority} | Qty: ${o.quantity}`
   435→        ).join('\n')
   436→        return { result: `Found ${orders.length} order(s):\n${summary}` }
   437→      } catch {
   438→        return { result: 'Could not fetch orders at this time.' }
   439→      }
   440→    }
   441→
   442→    case 'list_blueprints': {
   443→      try {
   444→        const params = new URLSearchParams()
   445→        if (args.status) params.set('status', args.status as string)
   446→        const res = await fetch(`${baseUrl}/api/blueprints?${params}`)
   447→        const data = await res.json()
   448→        const blueprints = (Array.isArray(data) ? data : data.blueprints ?? [])
   449→          .slice(0, (args.limit as number) ?? 5)
   450→        if (blueprints.length === 0) return { result: 'No blueprints found matching the criteria.' }
   451→        const summary = blueprints.map((b: Record<string, unknown>) =>
   452→          `${b.name} | Category: ${b.category} | Material: ${b.material} | Status: ${b.status} | Prints: ${b.printCount ?? 0}`
   453→        ).join('\n')
   454→        return { result: `Found ${blueprints.length} blueprint(s):\n${summary}` }
   455→      } catch {
   456→        return { result: 'Could not fetch blueprints at this time.' }
   457→      }
   458→    }
   459→
   460→    case 'create_order': {
   461→      try {
   462→        const body = {
   463→          partName: args.partName,
   464→          priority: args.priority ?? 'normal',
   465→          quantity: args.quantity ?? 1,
   466→          notes: args.notes ?? '',
   467→          status: 'pending',
   468→        }
   469→        const res = await fetch(`${baseUrl}/api/orders`, {
   470→          method: 'POST',
   471→          headers: { 'Content-Type': 'application/json' },
   472→          body: JSON.stringify(body),
   473→        })
   474→        const data = await res.json()
   475→        const id = data.id ?? data.orderId ?? 'NEW'
   476→        return {
   477→          result: `Order created successfully. ID: ${id}, Part: ${args.partName}, Priority: ${args.priority}, Qty: ${args.quantity}`,
   478→          frontendAction: {
   479→            type: 'created',
   480→            entity: 'order',
   481→            label: `Order created: ${args.partName} (${args.priority} priority)`,
   482→            id,
   483→          },
   484→        }
   485→      } catch {
   486→        return { result: 'Could not create order at this time.' }
   487→      }
   488→    }
   489→
   490→    case 'list_materials': {
   491→      try {
   492→        const res = await fetch(`${baseUrl}/api/materials`)
   493→        const data = await res.json()
   494→        let materials = Array.isArray(data) ? data : data.materials ?? []
   495→        if (args.status) materials = materials.filter((m: Record<string, unknown>) => m.status === args.status)
   496→        materials = materials.slice(0, 6)
   497→        if (materials.length === 0) return { result: 'No materials found matching the criteria.' }
   498→        const summary = materials.map((m: Record<string, unknown>) =>
   499→          `${m.name} | Category: ${m.category} | Stock: ${m.stockLevel ?? m.currentStock} | Status: ${m.status}`
   500→        ).join('\n')
   501→        return { result: `Found ${materials.length} material(s):\n${summary}` }
   502→      } catch {
   503→        return { result: 'Could not fetch materials at this time.' }
   504→      }
   505→    }
   506→
   507→    case 'list_certifications': {
   508→      try {
   509→        const res = await fetch(`${baseUrl}/api/certifications`)
   510→        const data = await res.json()
   511→        let certs = Array.isArray(data) ? data : data.certifications ?? []
   512→        if (args.status) certs = certs.filter((c: Record<string, unknown>) => c.status === args.status)
   513→        certs = certs.slice(0, 6)
   514→        if (certs.length === 0) return { result: 'No certifications found matching the criteria.' }
   515→        const summary = certs.map((c: Record<string, unknown>) =>
   516→          `${c.name} | Issuer: ${c.issuer} | Status: ${c.status} | Expires: ${c.expiryDate ?? 'N/A'}`
   517→        ).join('\n')
   518→        return { result: `Found ${certs.length} certification(s):\n${summary}` }

   520→        return { result: 'Could not fetch certifications at this time.' }

   522→    }

   524→    case 'get_risk_scores': {

   526→        const threshold = (args.threshold as number) ?? 70
   527→        const limit = (args.limit as number) ?? 5

   529→        const data = await res.json()
   530→        let parts = Array.isArray(data) ? data : data.parts ?? []
   531→        parts = parts.filter((p: Record<string, unknown>) => (p.riskScore as number) > threshold).slice(0, limit)
   532→        if (parts.length === 0) return { result: `No parts found with risk score above ${threshold}.` }
   533→        const summary = parts.map((p: Record<string, unknown>) =>
   534→          `${p.partName} (ID: ${p.id}) | Risk: ${p.riskScore}/100 | Lead Time: ${p.leadTimeDays}d | Stock: ${p.currentStock}/${p.minStock}`

   538→        return { result: 'Could not fetch risk scores at this time.' }
   539→      }
   540→    }
   541→
   542→    case 'predict_stockout': {
   543→      try {
   544→        const material = args.material as string
   545→        const centerId = args.center_id as string
   546→        const res = await fetch(`${baseUrl}/api/materials?name=${encodeURIComponent(material)}&center=${centerId}`)
   547→        const data = await res.json()

   550→        const daysRemaining = mat.daysUntilStockout ?? Math.ceil((mat.currentStock ?? 0) / (mat.consumptionRate ?? 1))
   551→        return { result: `Material "${material}" at center ${centerId}: ${daysRemaining} days until stockout. Current stock: ${mat.currentStock}/${mat.safetyStock}.` }
   552→      } catch {

   656→function detectNavigation(message: string): { page: string; label: string } | null {
   657→  const lower = message.toLowerCase()
   658→  // explicit "go to" / "show me" / "open" / "navigate" phrasing
   659→  const hasNavVerb = /\b(go to|goto|open|show me|take me to|navigate to|view|see)\b/.test(lower)
   660→  let bestPage: string | null = null
   661→  let bestScore = 0
   662→  for (const [page, keywords] of Object.entries(PAGE_KEYWORDS)) {
   663→    for (const kw of keywords) {
   664→      if (lower.includes(kw)) {
   665→        // longer keyword matches are more specific
   666→        const score = kw.length
   667→        if (score > bestScore) {
   668→          bestScore = score
   669→          bestPage = page
   670→        }
   671→      }
   672→    }
   673→  }
   674→  if (!bestPage) return null
   675→  // If there's no nav verb, only navigate on very specific single-word page mentions
   676→  // like "show orders" / "go to analytics". For ambiguous data questions like
   677→  // "how many orders" we still want to navigate if the user clearly wants the page.
   678→  const labels: Record<string, string> = {
   679→    overview: 'Dashboard Overview', orders: 'Orders', print_queue: 'Print Queue',
   680→    blueprints: 'Blueprint Library', centers: 'Print Centers', peer_printers: 'Peer Printers',
   681→    shipments: 'Shipments', materials: 'Materials', partners: 'OEM Partners',
   682→    analytics: 'Analytics', audit: 'Audit Logs', certifications: 'Certifications',
   683→    authorities: 'Certification Authorities', services: 'Customer Success',
   684→    digital_inventory: 'Digital Inventory', physical_inventory: 'Physical Inventory',
   685→    ip_library: 'IP Library', lab_portal: 'Lab & Testing Portal', settings: 'Settings',
   686→    emergency: 'Emergency Response', oem_portal: 'OEM Self-Service Portal',
   687→    cooperative: 'Digital Cooperative', material_properties: 'Material Properties Library',
   688→    feasibility: 'AM Feasibility Triage', sc_intelligence: 'Supply Chain Intelligence',
   689→  }
   690→  void hasNavVerb
   691→  return { page: bestPage, label: labels[bestPage] ?? bestPage }
   692→}

   694→function detectDataFetches(message: string): string[] {
   695→  const lower = message.toLowerCase()
   696→  const tools: string[] = []
   697→  if (/\b(how many|status of|list of|show|current|latest|stats?|kpi|overview|summary)\b/.test(lower) || /\border/.test(lower)) {
   698→    if (/\border/.test(lower)) tools.push('list_orders')
   699→  }
   700→  if (/\bblueprint/.test(lower)) tools.push('list_blueprints')
   701→  if (/\bmaterial/.test(lower) && /\b(stock|level|status|inventory|low|critical)\b/.test(lower)) tools.push('list_materials')
   702→  if (/\b(certification|cert|certificate)/.test(lower)) tools.push('list_certifications')
   703→  if (/\b(stat|kpi|overview|dashboard|metric)/.test(lower)) tools.push('get_dashboard_stats')
   704→  return [...new Set(tools)]
   705→}

   709→export async function POST(req: NextRequest) {
   710→  try {
   711→    const ip = req.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ?? 'unknown'
   712→    if (!checkRateLimit(ip)) {
   713→      return NextResponse.json({ error: 'Too many requests. Please wait before trying again.' }, { status: 429 })
   714→    }
   715→
   716→    const { messages, role } = await req.json()
   717→    if (!messages || !Array.isArray(messages)) {
   718→      return NextResponse.json({ error: 'Invalid request' }, { status: 400 })
   719→    }
   720→
   721→    const baseUrl = new URL(req.url).origin
   722→    const roleCtx = ROLE_CONTEXT[role ?? 'admin'] ?? ROLE_CONTEXT.admin
   723→
   724→    // Find the latest user message for intent detection
   725→    const lastUser = [...messages].reverse().find((m: { role: string }) => m.role === 'user')
   726→    const userText: string = lastUser?.content ?? ''
   727→
   728→    // Detect navigation + data-fetch intents
   729→    const navIntent = detectNavigation(userText)
   730→    const dataTools = detectDataFetches(userText)
   731→
   732→    // Execute data-fetch tools to ground the LLM in live data
   733→    const contextBlocks: string[] = []
   734→    for (const toolName of dataTools) {
   735→      const { result } = await executeTool(toolName, {}, baseUrl)
   736→      contextBlocks.push(`[LIVE DATA — ${toolName}]\n${result}`)
   737→    }
   738→
   739→    const liveContext = contextBlocks.length
   740→      ? `\n\n--- LIVE DASHBOARD DATA (use this when answering) ---\n${contextBlocks.join('\n\n')}\n--- END LIVE DATA ---\n`
   741→      : ''
   742→
   743→    const fullSystemPrompt = `${SYSTEM_PROMPT}\n\n${roleCtx}${liveContext}\n\nIf the user asked to go to or view a specific page, you are navigating them there now${navIntent ? ` (→ ${navIntent.label})` : ''}. Mention it briefly in your reply.`
   744→
   745→    // Build conversation for the SDK. The SDK uses role 'assistant' for the
   746→    // system prompt and 'user'/'assistant' for the turns.
   747→    const sdkMessages: { role: string; content: string }[] = [
   748→      { role: 'assistant', content: fullSystemPrompt },
   749→      ...messages.slice(-10).map((m: { role: string; content: string }) => ({
   750→        role: m.role === 'assistant' ? 'assistant' : 'user',
   751→        content: typeof m.content === 'string' ? m.content : JSON.stringify(m.content),
   752→      })),
   753→    ]
   754→
   755→    // Call the LLM via z-ai-web-dev-sdk
   756→    let reply = ''
   757→    try {
   758→      // Dynamically import so the SDK (backend-only) is never pulled into client bundles
   759→      const ZAIModule = await import('z-ai-web-dev-sdk')
   760→      const ZAI = (ZAIModule as { default: { create: () => Promise<unknown> } }).default
   761→      const zai = await ZAI.create()
   762→
   763→      // Race the LLM call against a 25s timeout so the route ALWAYS responds,
   764→      // even if the upstream model hangs. Without this, a slow/hung LLM call
   765→      // would let the HTTP request hang until the client times out (HTTP 000).
   766→      const LLM_TIMEOUT_MS = 25_000
   767→      const timeoutPromise = new Promise<never>((_, reject) =>
   768→        setTimeout(() => reject(new Error('LLM_TIMEOUT')), LLM_TIMEOUT_MS),
   769→      )
   770→      const completion = await Promise.race([
   771→        (zai as {
   772→          chat: {
   773→            completions: {
   774→              create: (opts: { messages: { role: string; content: string }[]; thinking: { type: string } }) =>
   775→                Promise<{ choices: { message: { content?: string } }[] }>
   776→            }
   777→          }
   778→        }).chat.completions.create({
   779→          messages: sdkMessages,
   780→          thinking: { type: 'disabled' },
   781→        }),
   782→        timeoutPromise,
   783→      ])
   784→      reply = completion.choices?.[0]?.message?.content ?? ''
   785→    } catch (err) {
   786→      console.error('AI SDK error:', err)
   787→      // Graceful fallback so the assistant still responds if the LLM is unavailable
   788→      reply = navIntent
   789→        ? `I've navigated you to ${navIntent.label}. ${dataTools.length ? 'I also pulled the latest data for you.' : "Let me know what you'd like to do next."}`
   790→        : "I'm here to help with your AddManuChain dashboard. I can navigate you to any page, pull live order/blueprint/material data, or create new orders. What would you like to do?"
   791→    }
   792→
   793→    if (!reply || !reply.trim()) {
   794→      reply = "I'm here to help with your AddManuChain dashboard. What would you like to do?"
   795→    }
   796→
   797→    const frontendAction = navIntent
   798→      ? { type: 'navigate', page: navIntent.page, label: navIntent.label }
   799→      : undefined
   800→
   801→    return NextResponse.json({ reply, frontendAction })
   802→
   803→  } catch (err) {
   804→    console.error('AI chat error:', err)
   805→    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
   806→  }
   807→}
